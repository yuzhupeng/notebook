# MD5

## 长度

首先MD5加密数据的长度

N*512+448+64 = (N+1)*512 ( bit,N>=0 )

其中我们的输入存在于N*512中

448包含了输入的部分和填充的部分

64 代表了输入的长度

其实也就是, 加密会分组, 分组的长度是64字节

然后我么输入的长度必须满足 (填充后的长度)

len%64=56 (byte)

```
void MD5_Update(MD5_CTX* context, unsigned char* input, unsigned int inputlen)
{
    1.对传入的长度做一个记录
    2.对长度做一个判断,期望能够加密的长度>=64字节,
    如果长度>=64:
        做一个运算
    否则:
        数据复杂积累
}
```

```
void MD5_Final(MD5_CTX* context, unsigned char digest[16])
{
    对传入的数据填充,让其满足N*512+448+64 = (N+1)*512
    先满足448,然后调用MD5_Update
    然后满足64,然后调用MD5_Update
}
```

```c
    DWORD i; // [esp+4Ch] [ebp-24h]
    CHAR sztmp[4]; // [esp+50h] [ebp-20h] BYREF
    BYTE output[16]; // [esp+54h] [ebp-1Ch] BYREF
    DWORD pdwDataLen; // [esp+64h] [ebp-Ch] BYREF
    HCRYPTHASH phHash; // [esp+68h] [ebp-8h] BYREF

    if (!CryptAcquireContextA(&phProv, 0, 0, 1u, 0xF0000000))
    {
        return 0;
    }
    if (CryptCreateHash(phProv, 0x8003u, 0, 0, &phHash))
    {
        if (CryptHashData(phHash, pbData, dwDataLen, 0))
        {
            CryptGetHashParam(phHash, 2u, output, &pdwDataLen, 0);
            *lpString1 = 0;
            for (i = 0; i < pdwDataLen; ++i)
            {
                wsprintfA(sztmp, "%02X", output[i]);
                lstrcatA(lpString1, sztmp);
            }
            CryptDestroyHash(phHash);
            CryptReleaseContext(phProv, 0);
            return 1;
        }
        else
        {
            CryptDestroyHash(phHash);
            CryptReleaseContext(phProv, 0);
            return 0;
        }
    }
    else
    {
        CryptReleaseContext(phProv, 0);
        return 0;
    }
```

# SHA1

和MD5类似

它每次加密64字节
加密的东西其实和MD5类似
64*n + 56 + 8
56是多余的,也就是剩余的数据一定要在56以内
如果不是,就扩展为64再加密,直到剩余字节数在56以内

8是我们的长度信息
然后56+8=64,实现最后一次加密
未处理的数据都是0填充

- 代码C
    
    ```c
    #include<stdio.h>
    #include<stdlib.h>
    #include<string.h> 
    #include<windows.h>
    
    #define frol(x,y) ((x<<y)|(x>>(32-y)))  //循环左移 
    
    //一次循环过程，str为填充后的数据或是数据中的一部分 
    
    //一次性加密16字节
    void renc64(BYTE input[64], DWORD32 abcde[5])
    {
    	DWORD32 a, b, c, d, e;
    	DWORD32 dw_tmp, dw_arr[80];
    	int i;
    	
    	for (i = 0; i < 16; i++)
    	{
    		dw_arr[i] = (input[i * 4 + 0] << 24) |
    			(input[i * 4 + 1] << 16) |
    			(input[i * 4 + 2] << 8) |
    			(input[i * 4 + 3] << 0);
    	}
    	for (i = 16; i < 80; i++)
    	{
    		dw_tmp = dw_arr[i - 3] ^ dw_arr[i - 8] ^ dw_arr[i - 14] ^ dw_arr[i - 16];
    		dw_arr[i] = frol(dw_tmp, 1);
    	}
    
    	a = abcde[0];
    	b = abcde[1];
    	c = abcde[2];
    	d = abcde[3];
    	e = abcde[4];
    
    	for (i = 0; i < 80; i++)
    	{
    		dw_tmp = dw_arr[i];
    		switch (i / 20)
    		{
    		case 0:
    			dw_tmp += frol(a, 5) + ((b & c) | (d & ~b)) + e;
    			dw_tmp += 0x5a827999;
    			break;
    		case 1:
    			dw_tmp += frol(a, 5) + (b ^ c ^ d) + e;
    			dw_tmp += 0x6ed9eba1;
    			break;
    		case 2:
    			dw_tmp += frol(a, 5) + ((b & c) | (b & d) | (c & d)) + e;
    			dw_tmp += 0x8f1bbcdc;
    			break;
    		case 3:
    			dw_tmp += frol(a, 5) + (b ^ c ^ d) + e;
    			dw_tmp += 0xca62c1d6;
    			break;
    		}
    		e = d;
    		d = c;
    		c = frol(b, 30);
    		b = a;
    		a = dw_tmp;
    	}
    	abcde[0] += a;
    	abcde[1] += b;
    	abcde[2] += c;
    	abcde[3] += d;
    	abcde[4] += e;
    	return;
    }
    
    //sha-1算法 
    void SHA1(BYTE* input, long long len, BYTE* output)
    {
    	int i = 0;
    
    	BYTE b64Arr[64];
    	DWORD32 h[5] = { 0x67452301,0xefcdab89,0x98badcfe,0x10325476,0xc3d2e1f0 };
    	DWORD32 n = len, dw_tmp;
    
    	//SHA1 update
    	while (n >= 64)
    	{
    		memcpy(b64Arr, input + len - n, 64);
    		renc64(b64Arr, h);
    		n -= 64;
    	}
    
    	//SHA1 final
    	memset(b64Arr, 0, 64);
    	memcpy(b64Arr, input + len - n, n);//输入的剩余数据填充
    
    	b64Arr[n] = 0x80;
    
    	if (n >= 56)//56+8=64
    	{
    		renc64(b64Arr, h);
    		memset(b64Arr, 0, 64);
    	}
    	//剩余的数据在56字节以内,不管是0或者是其它数据
    	for (i = 56; i < 64; i++)
    	{
    		b64Arr[i] = ((len * 8) >> (63 - i) * 8) & 0xff;
    	}
    	renc64(b64Arr, h);
    	for (i = 0; i < 20; i++)
    	{
    		dw_tmp = (h[i / 4] >> ((3 - i % 4) * 8)) & 0xff;
    		sprintf(output + 2 * i, "%02X", dw_tmp);
    	}
    
    	return;
    }
    
    //测试 
    
    int main()
    {
    	BYTE input[] = "Wrs29KlQaSd2OuvVRfoT9lodWGRTCyWnnjR3vjMo5aaVBDWyfE2JPQYgLnmd";
    	BYTE output[64] = { 0 };
    	int len = strlen(input);
    	SHA1(input, len, output);
    	printf("%s\n", output);
    	return 0;
    }
    ```

    ```c
    #include <stdio.h>
    #include <stdlib.h>
    #include <string.h>
    #include <memory.h>
    #include <stddef.h>
    #include <windows.h>
    
    #define CH(x,y,z) (((x) & (y)) ^ (~(x) & (z)))
    #define MAJ(x,y,z) (((x) & (y)) ^ ((x) & (z)) ^ ((y) & (z)))
    #define EP0(x) (ROTRIGHT(x,2) ^ ROTRIGHT(x,13) ^ ROTRIGHT(x,22))
    #define EP1(x) (ROTRIGHT(x,6) ^ ROTRIGHT(x,11) ^ ROTRIGHT(x,25))
    #define SIG0(x) (ROTRIGHT(x,7) ^ ROTRIGHT(x,18) ^ ((x) >> 3))
    #define SIG1(x) (ROTRIGHT(x,17) ^ ROTRIGHT(x,19) ^ ((x) >> 10))
    #define ROTLEFT(a,b) (((a) << (b)) | ((a) >> (32-(b))))
    #define ROTRIGHT(a,b) (((a) >> (b)) | ((a) << (32-(b))))
    
    #define SHA256_BLOCK_SIZE 32  
    
    typedef struct 
    {
    	BYTE data[64];
    	DWORD datalen;
    	DWORD64 bitlen;
    	DWORD state[8];
    } SHA256_CTX;
    
    void sha256_init(SHA256_CTX* ctx);
    void sha256_update(SHA256_CTX* ctx, const BYTE data[], size_t len);
    void sha256_final(SHA256_CTX* ctx, BYTE hash[]);

    void sha256_transform(SHA256_CTX* ctx, const BYTE data[])
    {
    	DWORD a, b, c, d, e, f, g, h, i, j, t1, t2, m[64];
    
    	// initialization 
    	for (i = 0, j = 0; i < 16; ++i, j += 4)
    		m[i] = (data[j] << 24) | (data[j + 1] << 16) | (data[j + 2] << 8) | (data[j + 3]);
    	for (; i < 64; ++i)
    		m[i] = SIG1(m[i - 2]) + m[i - 7] + SIG0(m[i - 15]) + m[i - 16];
    
    	a = ctx->state[0];
    	b = ctx->state[1];
    	c = ctx->state[2];
    	d = ctx->state[3];
    	e = ctx->state[4];
    	f = ctx->state[5];
    	g = ctx->state[6];
    	h = ctx->state[7];
    
    	for (i = 0; i < 64; ++i) {
    		t1 = h + EP1(e) + CH(e, f, g) + k[i] + m[i];
    		t2 = EP0(a) + MAJ(a, b, c);
    		h = g;
    		g = f;
    		f = e;
    		e = d + t1;
    		d = c;
    		c = b;
    		b = a;
    		a = t1 + t2;
    	}
    
    	ctx->state[0] += a;
    	ctx->state[1] += b;
    	ctx->state[2] += c;
    	ctx->state[3] += d;
    	ctx->state[4] += e;
    	ctx->state[5] += f;
    	ctx->state[6] += g;
    	ctx->state[7] += h;
    }
    
    void sha256_init(SHA256_CTX* ctx)
    {
    	ctx->datalen = 0;
    	ctx->bitlen = 0;
    	ctx->state[0] = 0x6a09e667;
    	ctx->state[1] = 0xbb67ae85;
    	ctx->state[2] = 0x3c6ef372;
    	ctx->state[3] = 0xa54ff53a;
    	ctx->state[4] = 0x510e527f;
    	ctx->state[5] = 0x9b05688c;
    	ctx->state[6] = 0x1f83d9ab;
    	ctx->state[7] = 0x5be0cd19;
    }
    
    void sha256_update(SHA256_CTX* ctx, const BYTE data[], size_t len)
    {
    	DWORD i;
    
    	for (i = 0; i < len; ++i) {
    		ctx->data[ctx->datalen] = data[i];
    		ctx->datalen++;
    		if (ctx->datalen == 64) 
    		{
    			// 64 byte = 512 bit  means the buffer ctx->data has fully stored one chunk of message
    			// so do the sha256 hash map for the current chunk
    			sha256_transform(ctx, ctx->data);
    			ctx->bitlen += 512;
    			ctx->datalen = 0;
    		}
    	}
    }
    
    void sha256_final(SHA256_CTX* ctx, BYTE hash[])
    {
    	DWORD i;
    
    	i = ctx->datalen;
    
    	// Pad whatever data is left in the buffer.
    	if (ctx->datalen < 56) {
    		ctx->data[i++] = 0x80;  // pad 10000000 = 0x80
    		while (i < 56)
    			ctx->data[i++] = 0x00;
    	}
    	else {
    		ctx->data[i++] = 0x80;
    		while (i < 64)
    			ctx->data[i++] = 0x00;
    		sha256_transform(ctx, ctx->data);
    		memset(ctx->data, 0, 56);
    	}
    
    	// Append to the padding the total message's length in bits and transform.
    	ctx->bitlen += ctx->datalen * 8;
    	ctx->data[63] = ctx->bitlen;
    	ctx->data[62] = ctx->bitlen >> 8;
    	ctx->data[61] = ctx->bitlen >> 16;
    	ctx->data[60] = ctx->bitlen >> 24;
    	ctx->data[59] = ctx->bitlen >> 32;
    	ctx->data[58] = ctx->bitlen >> 40;
    	ctx->data[57] = ctx->bitlen >> 48;
    	ctx->data[56] = ctx->bitlen >> 56;
    	sha256_transform(ctx, ctx->data);
    
    	// copying the final state to the output hash(use big endian).
    	for (i = 0; i < 4; ++i) {
    		hash[i] = (ctx->state[0] >> (24 - i * 8)) & 0x000000ff;
    		hash[i + 4] = (ctx->state[1] >> (24 - i * 8)) & 0x000000ff;
    		hash[i + 8] = (ctx->state[2] >> (24 - i * 8)) & 0x000000ff;
    		hash[i + 12] = (ctx->state[3] >> (24 - i * 8)) & 0x000000ff;
    		hash[i + 16] = (ctx->state[4] >> (24 - i * 8)) & 0x000000ff;
    		hash[i + 20] = (ctx->state[5] >> (24 - i * 8)) & 0x000000ff;
    		hash[i + 24] = (ctx->state[6] >> (24 - i * 8)) & 0x000000ff;
    		hash[i + 28] = (ctx->state[7] >> (24 - i * 8)) & 0x000000ff;
    	}
    
    }
    ```
